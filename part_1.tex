\subsection{}

The velocity triangle at the blade tip is constructed for a tip speed ratio $\lambda = 4$ and an axial induction factor $a = 0.3$. First, the tangential induction factor $a'$ is computed from Eq.~(2):
%
\begin{equation*}
    a' = \frac{1 - 3a}{4a - 1} = \frac{1 - 3 \times 0.3}{4 \times 0.3 - 1} = \frac{0.1}{0.2} = 0.5
\end{equation*}

The axial velocity component at the rotor plane is:
%
\begin{equation*}
    V_0(1-a) = V_0(1 - 0.3) = 0.7\, V_0
\end{equation*}

The tip speed is obtained from the definition of the tip speed ratio (Eq.~6):
%
\begin{equation*}
    \lambda = \frac{\omega R}{V_0} \quad \Longrightarrow \quad \omega R = \lambda\, V_0 = 4\, V_0
\end{equation*}

Including the tangential induction, the rotational velocity component becomes:
%
\begin{equation*}
    \omega R(1+a') = 4\, V_0(1 + 0.5) = 6\, V_0
\end{equation*}

These two components define the velocity triangle shown in Figure~\ref{fig:q2}. The axial component $V_0(1-a) = 0.7\,V_0$ is directed perpendicular to the rotor plane, while the rotational component $\omega R(1+a') = 6\,V_0$ is directed along the rotor plane. The relative wind speed $V_{rel}$ is the vector sum of the two. The drawing is at a 1:1 aspect ratio so that all angles are true.

\begin{figure}[ht]
    \centering
    \begin{tikzpicture}[scale=1.5, x=1cm, y=1cm]
        
        % 1. Define main coordinates for the velocity triangle
        % Physical values (in units of V_0): rotational = 6, axial = 0.7
        \def\scalefactor{5/6}  % drawing scale to keep figure compact
        \def\rotational{6}     % omega*R*(1+a') = 6 V_0
        \def\axial{0.7}        % V_0*(1-a) = 0.7 V_0

        % Twist angle
        \def\twistangle{-2.65}

        % 2. Draw the Rotor Plane
        \draw[] (4,0) -- (-6,0) node[below right] {Rotor Plane};

        % 3. Draw the Airfoil and Chord Line (Rotated by \beta)
        \begin{scope}[rotate=\twistangle]
            % Chord line
            \draw[dashed, gray] (-6, 0) node[above right, text=darkgray] {Chord Line} -- (5, 0) coordinate (Chord);
            
            % Generic airfoil shape
            \draw[thick, fill=gray!10] (0, 0) coordinate (leading)
                .. controls (0, 0.4) and (-2.7, 0.6) .. (-5.2, 0)  % Upper surface
                .. controls (-2.7, -0.1) and (0, -0.3) .. (0, 0); % Lower surface
        \end{scope}

        % 4. Draw the Velocity Triangle
        \draw[<-, very thick, blue] (leading) -- ++(\rotational*\scalefactor,0) coordinate (Rotational) node[midway, above] {$\omega R (1+a')$};
        \draw[<-, very thick, red] (leading) -- ++(0,-\axial*\scalefactor) node[above left] {$V_{0}(1-a)$};
        \draw[<-, very thick, green!60!black] (leading) -- ++(\rotational*\scalefactor,-\axial*\scalefactor) coordinate (Relative) node[right] {$V_{rel}$};
        \draw[dashed, thin, lightgray] (0,-\axial*\scalefactor) -- (Relative);
        \draw[dashed, thin, lightgray] (\rotational*\scalefactor,0) -- (Relative);
        
        % 5. Mark the Angles
        % Flow angle \phi (between rotor plane and relative wind)
        \pic [draw, "$\phi$", angle radius=4.5cm, angle eccentricity=1.1] {angle = Relative--leading--Rotational};
        
        % Twist angle \beta (between rotor plane and chord line)
        \pic [draw, "$\beta$", angle radius=6.5cm, angle eccentricity=1.05] {angle = Chord--leading--Rotational};
        
        % Angle of attack \alpha (between chord line and relative wind)
        \pic [draw, "$\alpha$", angle radius=6cm, angle eccentricity=1.05] {angle = Relative--leading--Chord};

    \end{tikzpicture}
    \caption{Sketch of wind turbine airfoil and velocity triangle (Aspect ratio 1:1).}
    \label{fig:q2}
\end{figure}
